cmake_minimum_required(VERSION 3.5)

project(haller-control VERSION 0.1 LANGUAGES CXX)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(QT NAMES Qt6 Qt5 REQUIRED COMPONENTS Widgets LinguistTools)
find_package(Qt${QT_VERSION_MAJOR} REQUIRED COMPONENTS Widgets Core Network)

set(PROJECT_SOURCES
        main.cpp
        MainWindow.cpp
        MainWindow.hpp
        MainWindow.ui
        UdpNode.cpp
        UdpNode.hpp
        resource.qrc
        ${TS_FILES}
)

if(${QT_VERSION_MAJOR} GREATER_EQUAL 6)
    qt_add_executable(haller-control
        MANUAL_FINALIZATION
        ${PROJECT_SOURCES}
    )
else()  
    add_executable(haller-control
            ${PROJECT_SOURCES}
    )
endif()

target_link_libraries(haller-control PRIVATE Qt${QT_VERSION_MAJOR}::Widgets)
target_link_libraries(haller-control PRIVATE Qt${QT_VERSION_MAJOR}::Core)
target_link_libraries(haller-control PRIVATE Qt${QT_VERSION_MAJOR}::Network)

set_target_properties(haller-control PROPERTIES
    MACOSX_BUNDLE_GUI_IDENTIFIER my.example.com
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

install(TARGETS haller-control
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)

if(QT_VERSION_MAJOR EQUAL 6)
    qt_finalize_executable(haller-control)
endif()
